// fixes lag issue regarding DamagedPlayer
// put things in AnimateHitmarker instead

event DamagedPlayer
{
    StopEvent AnimateHitmarker 0.0
    RunEvent AnimateHitmarker 0.0
}

event AnimateHitmarker
{
	Animate	Hitmarker 	Alpha		"255"		Linear 0.0 0.0
	Animate	Hitmarker 	Alpha		"0"		Linear 0.1 0.01
}

//BONUS HEALTH PULSE
event HudHealthBonusPulse
{

	Animate	PlayerStatusHealthValue fgcolor "115 220 220 255" Linear 0.0 0.0

	Animate	PlayerHealthBonusImage 	Alpha		"255"		Spline 0.0 0.5
	Animate	PlayerHealthBonusImage 	Alpha		"100"		Spline 0.5 0.5

	RunEvent HudHealthBonusPulseLoop	1.0
}
event HudHealthBonusPulseLoop
{
	RunEvent HudHealthBonusPulse 0.0
}
event HudHealthBonusPulseStop
{
	Animate	PlayerStatusHealthValue fgcolor "235 235 235 255" Linear 0.0 0.0
	Animate	PlayerHealthBonusImage 	Alpha		"0"		Spline 0.0 1.0

	StopEvent HudHealthBonusPulse 0.0
	StopEvent HudHealthBonusPulseLoop 0.0
}

//SPY DISGUISE
event HudSpyDisguiseChanged
{
	Animate PlayerStatusSpyOutlineImage		Alpha		"100"			Spline 0.0 0.2
	
	Animate PlayerStatusSpyOutlineImage		Position	"c-200 c-200"	Spline 0.0 0.2
	Animate PlayerStatusSpyOutlineImage		Size		"400 400"		Spline 0.0 0.2

	RunEvent HudSpyDisguiseHide	0.7
}
event HudSpyDisguiseHide
{
	Animate PlayerStatusSpyOutlineImage		Position	"3 413"			Spline 0.0 0.2
	Animate PlayerStatusSpyOutlineImage		Size		"55 55"			Spline 0.0 0.2
	
	Animate PlayerStatusSpyOutlineImage		Alpha		"0"				Spline 0.2 0.1
}
event HudSpyDisguiseFadeIn
{
	RunEvent HudSpyDisguiseChanged	0
	Animate PlayerStatusSpyImage			Alpha		"100"			Spline 0.9 0.1	
}
event HudSpyDisguiseFadeOut
{
	RunEvent HudSpyDisguiseChanged	0
	Animate PlayerStatusSpyImage			Alpha		"0"				Spline 0.9 0.1	
}

//FIX STOPPED ANIMATIONS
event DefaultAnimCompetitiveGame_RestoreChatWindow
{
	Animate HudChat ypos r240 Accel 0 0

	RunEvent HudHealthDyingPulseStop
	RunEvent HudHealthBonusPulseStop
	
	RunEvent HudLowAmmoPulseStop
}

//CHAT WINDOW
event DefaultAnimCompetitiveGame_LowerChatWindow
{
	Animate HudChat ypos r240 Accel 0 0
}
event DefaultAnimHudTournament_MoveChatWindow
{
	RunEvent CompetitiveGame_LowerChatWindow 0
	RunEvent CompetitiveGame_RestoreChatWindow 10
}
