//BONUS HEALTH PULSE
event HudHealthBonusPulse
{

	Animate	PlayerStatusHealthValue fgcolor "115 220 220 255" Linear 0.0 0.0

	Animate	PlayerHealthBonusImage 	Alpha		"255"		Spline 0.0 0.5
	Animate	PlayerHealthBonusImage 	Alpha		"100"		Spline 0.5 0.5

	RunEvent HudHealthBonusPulseLoop	1.0
}
event HudHealthBonusPulseLoop
{
	RunEvent HudHealthBonusPulse 0.0
}
event HudHealthBonusPulseStop
{
	Animate	PlayerStatusHealthValue fgcolor "235 235 235 255" Linear 0.0 0.0
	Animate	PlayerHealthBonusImage 	Alpha		"0"		Spline 0.0 1.0

	StopEvent HudHealthBonusPulse 0.0
	StopEvent HudHealthBonusPulseLoop 0.0
}
